## Overview

Implementare SPLiCE (Sparse Linear Concept Discovery), un metodo deterministico di sparse decomposition sul dizionario RadLex senza training. Risolve il problema di non-identificabilità (M-001) che affligge sia il baseline che Path A.

**Key difference**: Deterministico per costruzione, no seed, no training, CPU-only.

## Approach Justification

**Literature backing**: SPLiCE usa sparse coding diretto sul vocabolario RadLex, un approccio stabile e deterministico. La letteratura su sparse coding mostra che Orthogonal Matching Pursuit (OMP) fornisce decomposizioni stabili con exact L0 sparsity.

**Why this solves M-001**: Non-identificabilità negli SAE emerge dalla multiplità di soluzioni equivalenti nello spazio latente. SPLiCE vincola la decomposizione al vocabolario RadLex pre-esistente, eliminando l'ambiguità.

**Chosen algorithm**: OrthogonalMatchingPursuit (OMP) con `n_nonzero_coefs=k=32`, modality-gap correction, clamp post-hoc per non-negatività. Lasso(positive=True) come alternativa documentata.

def decompose_image(
    image_emb: torch.Tensor,      # (512,) L2-normalised
    vocab_emb: torch.Tensor,      # (V, 512) text embeddings
    gap: torch.Tensor | None,     # (512,) modality gap vector
    k: int = 32,
) -> torch.Tensor:
    """Return sparse coefficient vector (V,) via OMP.

    Solves: min ||emb_corrected - vocab_emb.T @ c||²  s.t. nnz(c) <= k, c >= 0

    Args:
        image_emb: Single image embedding (L2-normalised).
        vocab_emb: Vocabulary term embeddings (rows = terms).
        gap: Modality gap vector for correction. If None, skipped.
        k: Number of non-zero coefficients (L0 sparsity).

    Returns:
        Sparse coefficient vector (V,) with exactly k non-zero entries.
    """
    emb = image_emb.clone()
    if gap is not None:
        emb = emb - gap  # modality-gap correction

    # Orthogonal Matching Pursuit: exact L0 sparsity
    omp = OrthogonalMatchingPursuit(n_nonzero_coefs=k, fit_intercept=False)
    X = vocab_emb.numpy()  # (V, 512) — dictionary atoms as rows
    y = emb.numpy()        # (512,) — target signal
    omp.fit(X.T, y)        # X.T is (512, V); solve y ≈ X.T @ c

    coeffs = torch.from_numpy(omp.coef_).float()  # (V,)
    coeffs = coeffs.clamp(min=0)  # enforce non-negativity post-hoc
    return coeffs
```

def run(config: SpliCEConfig, test_embeddings: torch.Tensor,
        image_ids: list[str], vocab_terms: list[str]) -> list[dict]:
    """Decompose all test images; return per-image concept lists.

    Uses SAFE deserialization via utils.load_tensor() (weights_only=True).

    Args:
        config: SPLiCE configuration.
        test_embeddings: Test set image embeddings (N, 512).
        image_ids: Image identifiers (N,).
        vocab_terms: Vocabulary term strings (V,).

    Returns:
        List of per-image dicts with top_k_concepts + pseudo_report.
        Compatible with SAE sample_explanations.json schema.
    """
    from utils import load_tensor, ensure_dir
    import json

    vocab_emb = load_tensor(config.vocab_emb_path)  # SAFE: weights_only=True
    gap = None
    if config.use_gap_correction:
        gap = load_tensor(config.gap_path)  # SAFE

    results = []
    for emb, img_id in zip(test_embeddings, image_ids):
        coeffs = decompose_image(emb, vocab_emb, gap, k=config.k)
        top_k = coeffs.topk(config.k)
        concepts = [
            {"term": vocab_terms[idx], "coefficient": float(val)}
            for idx, val in zip(top_k.indices.tolist(), top_k.values.tolist())
        ]
        pseudo_report = "Findings suggest: " + ", ".join(
            c["term"] for c in concepts[:5]
        )
        results.append({
            "image_id": img_id,
            "top_k_concepts": concepts,
            "pseudo_report": pseudo_report,
        })

    ensure_dir(config.output_dir / "sample_explanations.json")
    output_path = config.output_dir / "sample_explanations.json"
    with open(output_path, "w") as f:
        json.dump(results, f, indent=2)

    return results
```

import pytest
import torch
import sys
sys.path.insert(0, "src/")
from concept_discovery.spliece import decompose_image

class TestSpliCE:
    def test_decompose_image_returns_k_nonzero(self):
        vocab_emb = torch.randn(1030, 512)
        vocab_emb = vocab_emb / vocab_emb.norm(dim=1, keepdim=True)
        image_emb = torch.randn(512)
        image_emb = image_emb / image_emb.norm()

        coeffs = decompose_image(image_emb, vocab_emb, gap=None, k=32)

        assert coeffs.shape == (1030,)
        assert (coeffs > 0).sum() == 32  # exactly k non-zero
        assert (coeffs >= 0).all()  # non-negative

    def test_gap_correction_changes_result(self):
        vocab_emb = torch.randn(100, 512).norm(dim=1, keepdim=True)
        image_emb = torch.randn(512).norm()
        gap = torch.randn(512)

        coeffs_no_gap = decompose_image(image_emb, vocab_emb, gap=None, k=10)
        coeffs_with_gap = decompose_image(image_emb, vocab_emb, gap=gap, k=10)

        assert not torch.allclose(coeffs_no_gap, coeffs_with_gap), \
            "Gap correction should change decomposition"
